function iqTest(numbers){ 
  let even = numbers.split(' ').slice(0,3).filter(num => num % 2 === 0).length;
  return numbers.split(' ').findIndex(num => (even < 2) ? num % 2 === 0 : num % 2 !== 0) + 1;
}
